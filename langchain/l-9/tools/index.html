import { ref } from 'vue'

interface Tool {
  name: string
  description: string
  schema: Record<string, string>
  mockCall: (input: string) => string
}

const tools: Tool[] = [
  {
    name: 'search',

    schema: { query: 'string - 搜索关键词' },

  },
  {
    name: 'calculator',
    description: '执行数学计算',
    schema: { expression: 'string - 数学表达式' },

      // 安全的简单计算模拟（仅支持数字和基本运算符）

      if (!safe) return '计算错误：仅支持数字和基本运算符 + - * / ( )'
      try {
        const result = new Function('return ' + input)()
        return `计算结果：${input} = ${result}`

        return '计算错误：请输入有效的数学表达式'
      }

原理

Tool 是 Agent 与外部世界交互的接口。每个工具定义名称、描述和参数 Schema，Agent 根据描述判断何时调用哪个工具。LangChain.js 推荐使用 tool 函数配合 Zod schema 定义工具，提供类型安全的方式。

处理流程

  1. 使用 tool 函数定义工具名称、描述和 Zod 参数 Schema。
  2. 实现工具的执行函数，接收参数并返回结果。
  3. 将工具注册到 Agent，Agent 在推理时自动选择和调用。

注意事项

  • 工具描述要清晰具体，说明适用场景和输入格式。
  • 工具执行应有超时和错误处理，避免阻塞 Agent 推理。
  • 参数 Schema 越精确，Agent 传参错误越少。
  • LangChain.js 推荐使用 tool 函数 + Zod schema，旧版 DynamicTool 已不推荐。
